from typing import Iterable


def replace_first(items: list) -> Iterable:
    if len(items) > 1:
        x = items[0]
        items.pop(0)
        items.append(x)
        return items
    else:
        return items


if __name__ == "__main__":

def max_digit(number: int) -> int:
