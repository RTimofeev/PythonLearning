from typing import Iterable


def remove_all_before(items: list, border: int) -> Iterable:
    try:
        items.index(border)
        for x in range(0, items.index(border)):
            items.pop(0)
    except:
        pass
    return items


if __name__ == '__main__':
    print(list(remove_all_before([1, 2, 3, 4, 5], 5 )))
